INITIAL_AUXILIAR  ->  INITIAL$

INITIAL   ->  lambda
LINE   ->  RULE_HEADER DISY

// Para aritmetica

EXPR -> TERM

TERM -> FACTOR DIV TERM
TERM -> FACTOR

FACTOR -> ADD NUM
FACTOR -> SUB NUM
FACTOR -> NUM
FACTOR -> LBRACKET NUM RBRACKET

// Desambiguacion de & y |

DISY   	->  CONJ

ELEM_FACTOR   	->  PRIM
ELEM_FACTOR 	-> 	ELEM_FACTOR T
ELEM_FACTOR 	->  rule
ELEM_FACTOR   	->  ELEM_FACTOR ^ NUM
